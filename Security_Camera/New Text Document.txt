sudo nano /etc/systemd/system/flask_app.service

[Unit]
Description=Flask Application
After=network.target

[Service]
User=pi  # Replace with the username that will run the service
WorkingDirectory=/path/to/your/application  # Replace with the path to your Flask application directory
ExecStart=/path/to/python /path/to/your/application/your_app.py  # Replace with the path to your Python interpreter and your Flask app script
Restart=always
RestartSec=3

[Install]
WantedBy=multi-user.target

sudo systemctl daemon-reload

sudo systemctl enable flask_app.service

sudo systemctl start flask_app.service
